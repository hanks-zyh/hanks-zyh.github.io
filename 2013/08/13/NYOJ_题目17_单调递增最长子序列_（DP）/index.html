NYOJ 题目17 单调递增最长子序列 （DP）

单调递增最长子序列

时间限制：3000 ms  |  内存限制：65535 KB
难度：4
描述
求一个字符串的最长递增子序列的长度
如：dabdbf最长递增子序列就是abdf，长度为4
输入
第一行一个整数0<n<20,表示有n个字符串要处理
随后的n行，每行有一个字符串，该字符串的长度不会超过10000
输出
输出字符串的最长递增子序列的长度
样例输入
3
aaa
ababc
abklmncdefg
样例输出
1
3
7



与之类似 f[i]数组存放 包括 i的 i个字母的最长递增子序列 长度，

f[0] = 1; 

f[1] = 2;  (递增子序列：1,4 )

f[2] = 3 ; (递增子序列：1,4,7 )

f[3] = 2; (递增子序列：1,2 )

f[4] = 3; (递增子序列：1,4,5 或 1,2,5)

f[5] = 4; (递增子序列：1,4,7,8或1,4,5,8或1,2,5,8)

f[6] = 3; (递增子序列：1,2,3)


最后求出f[]中最大值即为最长递增子序列

 
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
	int n,i,j,max,f[10010]; 
	char s[10010];
	// bool vis[500];
	scanf("%d",&n);
	getchar();
	while(n--){
		scanf("%s",s);
		
		
		for(i=1;i<strlen(s);i++) f[i] = 0;
		f[0]=1;
		for(i=1;i<strlen(s);i++)
		{
			max=0;
			for(j=0;j<i;j++){
				if(s[j]<s[i] &&  max<f[j]) {
					f[i] = f[j];
					max = f[j];
				}
			}
			f[i]++;
		}

		for(max=1,i=0;i<strlen(s);i++){
			if(max < f[i]) max = f[i];
		}
		printf("%d\n",max);
	}
	
	return 0;
}
